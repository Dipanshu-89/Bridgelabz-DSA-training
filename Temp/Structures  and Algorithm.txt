## PROBLEM 1: AVERAGE TEMPERATURE

class Avg{
static double find(int a[],int n){
int total=0;
for(int i=0;i<n;i++) total+=a[i];
return (double)total/n;
}
}


## PROBLEM 2: SEARCH ALGORITHMS

## Linear Search

class Linear{
static boolean find(String a[],String t){
for(int i=0;i<a.length;i++)
if(a[i].equals(t)) return true;
return false;
}
}

## Binary Search

class Binary{
static boolean find(String a[],String t){
int l=0,r=a.length-1;
while(l<=r){
int m=(l+r)/2;
if(a[m].equals(t)) return true;
else if(a[m].compareTo(t)<0) l=m+1;
else r=m-1;
}
return false;
}
}

## Hash Search

import java.util.*;
class Hash{
static boolean find(HashSet<String>s,String t){
return s.contains(t);
}
}

## PROBLEM 3: PREREQUISITES CHECK

## Nested Loop

class CheckA{
static boolean ok(int a[],int b[]){
for(int i=0;i<b.length;i++){
boolean f=false;
for(int j=0;j<a.length;j++){
if(b[i]==a[j]){f=true;break;}
}
if(!f) return false;
}
return true;
}
}

## PROBLEM 4: SORTING

## Bubble Sort

class Bubble{
static void sort(int a[]){
for(int i=0;i<a.length;i++)
for(int j=0;j<a.length-i-1;j++)
if(a[j]>a[j+1]){
int t=a[j];a[j]=a[j+1];a[j+1]=t;
}
}
}

## Merge Sort

class Merge{
static void merge(int a[],int l,int m,int r){
int n1=m-l+1,n2=r-m;
int L[]=new int[n1],R[]=new int[n2];
for(int i=0;i<n1;i++) L[i]=a[l+i];
for(int j=0;j<n2;j++) R[j]=a[m+1+j];
int i=0,j=0,k=l;
while(i<n1&&j<n2)
a[k++]=(L[i]<=R[j])?L[i++]:R;
while(i<n1) a[k++]=L[i++];
while(j<n2) a[k++]=R[j++];
}
static void sort(int a[],int l,int r){
if(l<r){
int m=(l+r)/2;
sort(a,l,m);
sort(a,m+1,r);
merge(a,l,m,r);
}
}
}

## Quick Sort

class Quick{
static int part(int a[],int l,int r){
int p=a[r],i=l-1;
for(int j=l;j<r;j++){
if(a[j]<p){
i++;
int t=a[i];a[i]=a[j];a[j]=t;
}
}
int t=a[i+1];a[i+1]=a[r];a[r]=t;
return i+1;
}
static void sort(int a[],int l,int r){
if(l<r){
int pi=part(a,l,r);
sort(a,l,pi-1);
sort(a,pi+1,r);
}
}
}

## Insertion Sort

class Insert{
static void sort(int a[]){
for(int i=1;i<a.length;i++){
int key=a[i],j=i-1;
while(j>=0&&a[j]>key){
a[j+1]=a[j];
j--;
}
a[j+1]=key;
}
}

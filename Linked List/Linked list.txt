import java.io.*;
class DirSize{
static long size(File f){
if(f.isFile()) return f.length();
long total=0;
File list[]=f.listFiles();
if(list!=null){
for(File x:list) total+=size(x);
}
return total;
}
public static void main(String[]args){
File f=new File("project");
System.out.println(size(f));
}
}

2. WORD SEARCH (BACKTRACKING)

---

class Word{
static int dx[]={-1,-1,-1,0,0,1,1,1};
static int dy[]={-1,0,1,-1,1,-1,0,1};

static boolean find(char g[][],String w,int i,int j,int k,boolean v[][]){
if(k==w.length()) return true;
if(i<0||j<0||i>=4||j>=4||v[i][j]||g[i][j]!=w.charAt(k)) return false;
v[i][j]=true;
for(int d=0;d<8;d++){
if(find(g,w,i+dx[d],j+dy[d],k+1,v)) return true;
}
v[i][j]=false;
return false;
}

public static void main(String[]args){
char g[][]={{'C','A','T','S'},{'O','R','E','A'},{'D','E','A','M'},{'E','L','L','S'}};
String w="DREAM";
boolean v[][]=new boolean[4][4];
for(int i=0;i<4;i++)
for(int j=0;j<4;j++)
if(find(g,w,i,j,0,v)){
System.out.println("Found");
return;
}
System.out.println("Not Found");
}
}

3. N-QUEENS (OPTIMIZED)

---

class Queen{
static int n=4;
static boolean col[]=new boolean[n];
static boolean d1[]=new boolean[2*n];
static boolean d2[]=new boolean[2*n];
static int a[]=new int[n];

static boolean solve(int r){
if(r==n){
for(int i=0;i<n;i++)
System.out.print(a[i]+" ");
System.out.println();
return true;
}
for(int c=0;c<n;c++){
if(!col[c]&&!d1[r-c+n]&&!d2[r+c]){
a[r]=c;
col[c]=d1[r-c+n]=d2[r+c]=true;
if(solve(r+1)) return true;
col[c]=d1[r-c+n]=d2[r+c]=false;
}
}
return false;
}

public static void main(String[]args){
solve(0);
}
}

class Queen2{

static boolean d2[]=new boolean[2*n];
static boolean block[][]={{false,false,false,false},{false,true,false,false},{false,false,false,false},{false,false,false,false}};

if(r==n){
for(int i=0;i<n;i++)
System.out.print(a[i]+" ");
System.out.println();
return true;
}
for(int c=0;c<n;c++){
if(block[r][c]) continue;
if(!col[c]&&!d1[r-c+n]&&!d2[r+c]){
a[r]=c;
col[c]=d1[r-c+n]=d2[r+c]=true;
if(solve(r+1)) return true;
col[c]=d1[r-c+n]=d2[r+c]=false;
}
}
return false;
}
